  </ui-drawer-trigger>
  <ui-drawer-content>

    </ui-drawer-header>

    </div>
    <ui-drawer-footer>
      <ui-button>Submit</ui-button>
      <ui-drawer-close><ui-button variant="outline">Cancel</ui-button></ui-drawer-close>
    </ui-drawer-footer>
  </ui-drawer-content>
</ui-drawer>
