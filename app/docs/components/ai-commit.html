Commit

Displays a git commit with hash, message, author info, file changes, and diff statistics.

Preview

a1b2c3d Fix authentication flow Jane Doe src/auth/login.ts

Usage

<ui-ai-commit>
  <ui-ai-commit-header>
    <ui-ai-commit-hash>a1b2c3d</ui-ai-commit-hash>
    <ui-ai-commit-message>Fix authentication flow</ui-ai-commit-message>
  </ui-ai-commit-header>
  <ui-ai-commit-metadata>
    <ui-ai-commit-info>
      <ui-ai-commit-author>
        <ui-ai-commit-author-avatar initials="JD"></ui-ai-commit-author-avatar>
        Jane Doe
      </ui-ai-commit-author>
      <ui-ai-commit-separator></ui-ai-commit-separator>
      <ui-ai-commit-timestamp date="2024-01-15T10:30:00Z"></ui-ai-commit-timestamp>
    </ui-ai-commit-info>
    <ui-ai-commit-actions>
      <ui-ai-commit-copy-button hash="a1b2c3d"></ui-ai-commit-copy-button>
    </ui-ai-commit-actions>
  </ui-ai-commit-metadata>
  <ui-ai-commit-content>
    <ui-ai-commit-files>
      <ui-ai-commit-file>
        <ui-ai-commit-file-info>
          <ui-ai-commit-file-status status="modified"></ui-ai-commit-file-status>
          <ui-ai-commit-file-path>src/auth/login.ts</ui-ai-commit-file-path>
        </ui-ai-commit-file-info>
        <ui-ai-commit-file-changes>
          <ui-ai-commit-file-additions count="12"></ui-ai-commit-file-additions>
          <ui-ai-commit-file-deletions count="3"></ui-ai-commit-file-deletions>
        </ui-ai-commit-file-changes>
      </ui-ai-commit-file>
    </ui-ai-commit-files>
  </ui-ai-commit-content>
</ui-ai-commit>

Sub-components

<ui-ai-commit> <ui-ai-commit-header> <ui-ai-commit-hash> <ui-ai-commit-message> <ui-ai-commit-metadata> <ui-ai-commit-separator> <ui-ai-commit-info> <ui-ai-commit-author> <ui-ai-commit-author-avatar> <ui-ai-commit-timestamp> <ui-ai-commit-actions> <ui-ai-commit-copy-button> <ui-ai-commit-content> <ui-ai-commit-files> <ui-ai-commit-file> <ui-ai-commit-file-info> <ui-ai-commit-file-status> <ui-ai-commit-file-icon> <ui-ai-commit-file-path> <ui-ai-commit-file-changes> <ui-ai-commit-file-additions> <ui-ai-commit-file-deletions>

Props

<ui-ai-commit-author-avatar>

PropTypeDefaultDescription
initialsstring—Author initials for the avatar.

<ui-ai-commit-timestamp>

PropTypeDefaultDescription
dateDate—The commit date.

<ui-ai-commit-copy-button>

PropTypeDefaultDescription
hashstring—The commit hash to copy.
onCopy() => void—Callback on copy.
onError(error: Error) => void—Callback on error.
timeoutnumber—Duration in ms to show success state.

<ui-ai-commit-file-status>

PropTypeDefaultDescription
status"added" | "modified" | "deleted" | "renamed"—The file change status.

<ui-ai-commit-file-additions>

PropTypeDefaultDescription
countnumber—Number of added lines.

<ui-ai-commit-file-deletions>

PropTypeDefaultDescription
countnumber—Number of deleted lines.

All components also accept standard HTML attributes (class, id, style, data-*, aria-*, etc.) which are passed through to the underlying element.
